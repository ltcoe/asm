org 100h

.data
str1 db "hello$"
str2 db 6 dup ('$')

msg1 db 10,13,'stored string is : $ '
msg2 db 10,13, 'reversed string is : $ '

Display macro msg
	mov ah, 9
	lea dx, msg
	int 21h
Endm

.code

mov ax, @data 
mov ds,ax
Display msg1
Display str1

lea si,str2
lea di,str1
add di, 4
mov cx, 5

Reverse:
mov al,[di]
mov [si], al
inc si
dec di
loop Reverse

Display msg2
Display str2

mov ah,4ch
int 21h

ret
